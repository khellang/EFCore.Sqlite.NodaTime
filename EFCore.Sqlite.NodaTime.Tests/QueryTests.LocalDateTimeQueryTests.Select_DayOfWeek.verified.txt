    Type: ReaderExecuted,
    Text: 
SELECT "n"."Id", "n"."Instant", "n"."LocalDate", "n"."LocalDateTime", "n"."LocalTime"
FROM "NodaTimeTypes" AS "n"
WHERE CASE
    WHEN CAST(strftime('%w', "n"."LocalDateTime") AS INTEGER) = 0 THEN 7
    ELSE CAST(strftime('%w', "n"."LocalDateTime") AS INTEGER)
END = 6
LIMIT 2
  }
